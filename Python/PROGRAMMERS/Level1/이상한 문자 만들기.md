# 이상한 문자 만들기

```python
def solution(s):
    answer = ''
    words = list(map(str, s))
    k = 1
    while words:
        word = words.pop(0)
        if k % 2 == 0:
            answer += word.lower()
        else:
            answer += word.upper()
        if word == ' ':
            k = 0
        k += 1
    return answer
```
