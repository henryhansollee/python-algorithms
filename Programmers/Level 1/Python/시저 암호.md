# 시저 암호

```python
def solution(s, n):
    answer = ''
    
    # a: 97, z: 122, A: 65, Z: 90
    upper_alphas = []
    lower_alphas = []
    for i in range(97, 123):
        upper_alphas.append(chr(i))
    for i in range(65, 91):
        lower_alphas.append(chr(i))
    
    for i in range(len(s)):
        if s[i] in upper_alphas:
            idx = upper_alphas.index(s[i])
            if idx+n < len(upper_alphas):
                answer += upper_alphas[idx+n]
            else:
                answer += upper_alphas[idx+n-len(upper_alphas)]
        if s[i] in lower_alphas:
            idx = lower_alphas.index(s[i])
            if idx+n < len(lower_alphas):
                answer += lower_alphas[idx+n]
            else:
                answer += lower_alphas[idx+n-len(lower_alphas)]
        if s[i] == ' ':
            answer += ' '
                
    return answer
```
