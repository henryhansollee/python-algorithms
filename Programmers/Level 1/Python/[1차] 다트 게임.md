# [1차] 다트 게임

```python
def solution(dartResult):
    answer = 0
    n = len(dartResult)
    
    temp = [0]
    num = ''
    for i in range(n):
        if dartResult[i].isdigit():
            num += dartResult[i]
        else:
            if dartResult[i] == 'S':
                temp.append(int(num) ** 1)
            elif dartResult[i] == 'D':
                temp.append(int(num) ** 2)
            elif dartResult[i] == 'T':
                temp.append(int(num) ** 3)
            elif dartResult[i] == '*':
                temp[-2] *= 2
                temp[-1] *= 2
            elif dartResult[i] == '#':
                temp[-1] *= -1
            num = ''
            
    answer = sum(temp)
    
    return answer
```
