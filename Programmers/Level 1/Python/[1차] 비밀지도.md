# [1차] 비밀지도

```python
def solution(n, arr1, arr2):
    answer = []
    n = len(arr1)
    
    s_arr1 = []
    s_arr2 = []
    
    for i in range(len(arr1)):
        temp1 = str(bin(arr1[i]))[2:]
        temp2 = str(bin(arr2[i]))[2:]
        if len(temp1) < n:
            temp1 = ('0' * (n-len(temp1))) + temp1
        if len(temp2) < n:
            temp2 = ('0' * (n-len(temp2))) + temp2
        s_arr1.append(list(temp1))
        s_arr2.append(list(temp2))
    
    for i in range(n):
        for j in range(n):
            if s_arr1[i][j] == '1' or s_arr2[i][j] == '1':
                s_arr1[i][j] = '#'
            if s_arr1[i][j] == '0' and s_arr2[i][j] == '0':
                s_arr1[i][j] = ' '
    
    for i in range(n):
        answer.append(''.join(s_arr1[i]))
    
    return answer

```
