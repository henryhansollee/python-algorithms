# 로또의 최고 순위와 최저 순위

```python
def solution(lottos, win_nums):
    n = len(lottos)
    
    cnt, zero = 0, 0
    for i in range(n):
        if lottos[i] in win_nums:
            cnt += 1
        if lottos[i] == 0:
            zero += 1

    max_rank = 7 - (cnt + zero)
    min_rank = 7 - cnt
    
    if max_rank >= 6:
        max_rank = 6
    if min_rank >= 6:
        min_rank = 6
        
    answer = [max_rank, min_rank]
    
    return answer
```
