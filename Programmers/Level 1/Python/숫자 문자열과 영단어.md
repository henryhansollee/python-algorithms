# 숫자 문자열과 영단어

```python
def solution(s):
    answer = ''
    n = len(s)
    
    nums = [
        'zero', 'one', 'two', 'three', 'four', 'five', 'six', 'seven', 'eight', 'nine'
    ]
    
    temp = ''
    for i in range(n):
        if s[i].isdigit():
            answer += s[i]
        else:
            temp += s[i]
        for j in range(10):
            if temp == nums[j]:
                answer += str(j)
                temp = ''
    
    return int(answer)
```
