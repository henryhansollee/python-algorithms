# 모의고사

```python
def solution(answers):
    answer = []
    
    n = len(answers)
    a = [1, 2, 3, 4, 5] * n
    b = [2, 1, 2, 3, 2, 4, 2, 5] * n
    c = [3, 3, 1, 1, 2, 2, 4, 4, 5, 5] * n
    
    a_answers, b_answers, c_answers = 0, 0, 0
    
    for i in range(n):
        if answers[i] == a[i]:
            a_answers += 1
        if answers[i] == b[i]:
            b_answers += 1
        if answers[i] == c[i]:
            c_answers += 1
    
    max_abc = max(a_answers, b_answers, c_answers)
    
    if max_abc == a_answers:
        answer.append(1)
    if max_abc == b_answers:
        answer.append(2)
    if max_abc == c_answers:
        answer.append(3)

    return answer
```
