# 완주하지 못한 선수

```python
def solution(participant, completion):
    answer = ''
    
    n = len(completion)
    
    participant.sort()
    completion.sort()
    
    flag = True
    for i in range(n):
        if participant[i] != completion[i]:
            answer = participant[i]
            flag = False
            break

    if flag:
        answer = participant[-1]
    
    return answer
```
