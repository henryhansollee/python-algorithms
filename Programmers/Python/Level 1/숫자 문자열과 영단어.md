# 숫자 문자열과 영단어

###### 나의 풀이

```python
def solution(s):
    answer = ''
    n = len(s)
    
    nums = [
        'zero', 'one', 'two', 'three', 'four', 'five', 'six', 'seven', 'eight', 'nine'
    ]
    
    temp = ''
    for i in range(n):
        if s[i].isdigit():
            answer += s[i]
        else:
            temp += s[i]
        for j in range(10):
            if temp == nums[j]:
                answer += str(j)
                temp = ''
    
    return int(answer)
```

###### 다른 풀이

```python
num_dic = {"zero":"0", "one":"1", "two":"2", "three":"3", "four":"4", "five":"5", "six":"6", "seven":"7", "eight":"8", "nine":"9"}

def solution(s):
    answer = s
    for key, value in num_dic.items():
        answer = answer.replace(key, value)
    return int(answer)
```
