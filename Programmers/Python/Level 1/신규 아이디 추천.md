# 신규 아이디 추천

###### 나의 풀이

```python
def solution(new_id):
    
    # 1단계, 소문자 치환
    new_id = new_id.lower()
    new_id = list(new_id)
    
    # 2단계, 특수문자 제거(숫자, chr(97~122), chr(45), chr(95), chr(46))
    new = []
    for i in range(len(new_id)):
        if new_id[i].isdigit() or 97 <= ord(new_id[i]) <= 122 or ord(new_id[i]) == 45 or ord(new_id[i]) == 95 or ord(new_id[i]) == 46:
                new.append(new_id[i])

    # 3단계, '.'가 2번 이상 연속되면 하나로
    stack = []
    while new:
        n = new.pop(0)
        if not stack:
            stack.append(n)
        else:
            if stack[-1] == '.' and n == '.': continue
            else:
                stack.append(n)
    
    # 4단계, '.'가 처음이나 끝이면 제거
    if stack[0] == '.':
        stack.pop(0)
    if stack and stack[-1] == '.':
        stack.pop()

    # 5단계, 비었으면 'a' 대입
    if not stack:
        stack.append('a')
        
    # 6단계, 16자 이상이면, 15빼고 제거
    stack = stack[:15]
    if stack and stack[-1] == '.':
        stack.pop()
        
    # 7단계, 2자 이하라면 마지막 문자를 길이 3이 될 때까지 반복
    while len(stack) < 3:
        stack.append(stack[-1])
                
    return ''.join(map(str, stack))
```

```python
import re

def solution(new_id):
    st = new_id
    st = st.lower()
    st = re.sub('[^a-z0-9\-_.]', '', st)
    st = re.sub('\.+', '.', st)
    st = re.sub('^[.]|[.]$', '', st)
    st = 'a' if len(st) == 0 else st[:15]
    st = re.sub('^[.]|[.]$', '', st)
    st = st if len(st) > 2 else st + "".join([st[-1] for i in range(3-len(st))])
    return st
```
