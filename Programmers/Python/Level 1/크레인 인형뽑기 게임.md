# 크레인 인형뽑기 게임

###### 나의 풀이

```python
def solution(board, moves):
    answer = 0
    
    stack = []
    cnt = 0
    for c in moves:
        for r in range(len(board)):
            if board[r][c-1] != 0:
                if len(stack) > 0:
                    if stack[-1] == board[r][c-1]:
                        board[r][c-1] = 0
                        stack.pop()
                        cnt += 1
                        break
                stack.append(board[r][c-1])
                board[r][c-1] = 0
                break
    answer = cnt * 2
    
    return answer
```

                if len(stacklist) > 1:
                    if stacklist[-1] == stacklist[-2]:
                        stacklist.pop(-1)
